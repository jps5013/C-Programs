#include <cs50.h>
#include <stdio.h>
#include <ctype.h>
#include <string.h>

int main(void)
{
    // Get user name
    string name = get_string();
    if (name !=  NULL)
    // no matter what print first initial
    printf("%c", toupper(name[0]));
    //loop over string to identify each character name[i]
    for(int i = 0, n = strlen(name); i < n ; i ++)
    {   
        if((char)name[i - 1] == ' ')
        {
            printf("%c", toupper(name[i]));
        }
    }
    printf("\n");
}
