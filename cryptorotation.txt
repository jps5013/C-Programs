char ASCIItoOffset(char c)
{
    int offset;
    if(islower(c))
    {
        offset = c - 'a';
    }
    else
    {
        offset = c - 'A';   
    }
    return offset;
}

char OffsettoASCII(int x, char forCase)
{   
    int reset; 
    if(islower(forCase))
    {
        reset = x + 'a'; 
    }
    else
    {
        reset = x + 'A';   
    }
    return reset;
}
